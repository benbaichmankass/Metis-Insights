# Backtest-only dependencies (S-TRAINER-BT-1, 2026-05-17).
#
# These are the deps the trainer-VM needs for the issue-driven backtest
# sweep (scripts/ops/run_backtest_sweep.sh + the experiments/ harnesses).
# pandas + numpy are already in requirements.txt (production runtime
# uses them); this file adds the parquet engine + the data-fetcher
# extras. Install via:
#
#     /home/ubuntu/.venvs/ict-trainer/bin/pip install -r requirements.txt -r requirements-backtest.txt
#
# Bootstrap script `scripts/ops/trainer_bootstrap.sh` does both files
# in one call. Runbook: docs/runbooks/trainer-backtest.md.

# Parquet engine — pandas uses it via DataFrame.to_parquet / read_parquet.
# The 2026-05-08 experiment harness hand-installed it on the trainer
# (no requirements file); pinning here so future trainer re-provisions
# get it deterministically.
pyarrow>=14.0.0

# HTTP fetcher for the qashdev/btc monthly CSV archive (Binance Vision
# mirror). The fetcher hits raw.githubusercontent.com.
requests>=2.31.0

# Used by the cache-freshness check (parsing month strings → datetimes).
# Already a transitive dep of pandas; pinning explicitly so the
# fetcher can `import dateutil` without lying about its deps.
python-dateutil>=2.8.2

# Pretrained-TSFM embeddings (M19 T0.1). `scripts/ml/build_embeddings.py` runs a
# frozen amazon/chronos-bolt-tiny (9M, CPU) as a feature extractor to produce the
# `tsfm_emb_*` side-stream. TRAINER-SIDE ONLY — the live money-box never installs
# these (T0.1 heads stay at `candidate`/offline; `ml.datasets.embedding_features`
# lazy-imports both, so `ml.*` imports fine without them). Torch is CPU-only on
# the ARM trainer. Promotion to shadow (embedding computed live) is the gated
# follow-up that would revisit the live-VM dependency.
chronos-forecasting>=1.4.0
torch>=2.2.0
