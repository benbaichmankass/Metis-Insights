# =============================================================================
# Production runtime dependencies for ict-trading-bot.
#
# Pinning policy:
#   - Floor pin (>=) per package. Major-version ceiling left open so
#     security patches land via `pip install -U`.
#   - apscheduler / pytz / tzlocal pinned together — see BUG-005:
#     apscheduler 3.6.3 + tzlocal 5.x crash-loop on Python 3.10 with a
#     timezone-format mismatch. Working set in production:
#       apscheduler 3.11.2 / tzlocal 5.3.1 / pytz 2026.1.post1
#     Floors below match what was hand-installed on the VM during the
#     S-014.5 fix; they prevent BUG-005 from re-occurring on a fresh
#     VM bootstrap.
#
# Test-only deps live in requirements-test.txt — avoid mixing.
# =============================================================================

# Gradient-boosted trees for the v2 model layer (LightGBMMulticlassTrainer +
# LightGBMRegressionTrainer). Lazy-imported inside the trainer/predictor so
# environments without lightgbm can still import ml.* — only `python -m ml
# train …` against a LightGBM manifest requires it.
# EXACT pin (2026-07-20, MB-20260720-LIVE-SERVING-PARITY-SKEW): the live and
# trainer venvs must run the SAME lightgbm — an unpinned floor let the two
# stacks diverge, producing ~1e-2 score skew between live-served and
# trainer-re-scored predictions from the SAME model file + SAME feature row
# (the M25 live_parity gate's 100% fresh-row mismatch). 4.6.0 = the trainer's
# current version. Bump BOTH venvs together, never one side alone.
lightgbm==4.6.0

# Firebase Cloud Messaging publishing (mobile push, M12 S1). Used by
# src/runtime/mobile_push/notifier.py to OAuth2-sign the service-account
# JSON for FCM HTTP v1. Optional at runtime — when absent, the notifier
# degrades to inert + logs a WARNING (the trader is unaffected).
google-auth>=2.0.0

# Tradovate WebSocket client (futures broker, PR #2647 + wiring PR).
# Imported inside src/units/accounts/tradovate/websocket_client.py behind a
# try/except so the rest of the Tradovate package (REST, models, risk) works
# without it — required only when streaming quotes / order events. The
# trader unit imports the package via the integrator only when an account
# carries `exchange: tradovate`, so absence is degrade-to-warning not a
# crash. Floor 12.x picked to match the websockets-12 API used in
# websocket_client.py (the deprecated `WebSocketClientProtocol` import path).
websockets>=12.0
